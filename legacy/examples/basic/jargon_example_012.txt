Code:
ASK "Enter a word" as text
SET freq ([])
SET i (0)

REPEAT_UNTIL i reaches end of text
    SET char (text[i])
    SET count (0)
    SET j (0)
    REPEAT_UNTIL j reaches end of text
        IF text[j] is equal to char THEN
            SET count (count + 1)
        END
        SET j (j + 1)
    END
    PRINT char
    PRINT count
    SET i (i + 1)
END
