SET nested ([[1, 2], [3, 4], [5, 6]])
SET total (0)
SET i (0)

REPEAT_UNTIL i reaches end of nested
    SET sublist (nested[i])
    SET j (0)
    REPEAT_UNTIL j reaches end of sublist
        IF sublist[j] is even THEN
            SET total (total + sublist[j])
        END
        SET j (j + 1)
    END
    SET i (i + 1)
END

PRINT total

# Expected Output:
# 12

# Explanation:
# Nested loop accumulates even values from a 2D list.
